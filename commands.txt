#!/usr/bin/env bash
set -Eeuo pipefail

VOLUME_SRC="/mnt/raid/volumes"
VOLUME_DST="/var/lib/docker/volumes"
APP_DATA_DIR="$HOME/datamancy"
QDRANT_DIR="/mnt/vectors/qdrant"

log() {
  printf '\n==> %s\n' "$*"
}

require() {
  command -v "$1" >/dev/null 2>&1 || {
    echo "Missing required command: $1" >&2
    exit 1
  }
}

cleanup_dir_contents() {
  local dir="$1"
  sudo mkdir -p "$dir"
  sudo find "$dir" -mindepth 1 -maxdepth 1 -exec rm -rf -- {} +
}

log "Checking required commands"
require docker
require systemctl
require mount
require findmnt
require find

log "Refreshing sudo"
sudo -v

log "Stopping and removing running containers"
docker ps -q | xargs -r docker stop -t 0 || true
docker ps -aq | xargs -r docker rm -f || true

log "Pruning buildx/buildkit state"
docker buildx prune -af || true
docker buildx rm --all-inactive || true
docker ps -aq --filter name=buildx_buildkit | xargs -r docker rm -f || true
docker ps -aq --filter name=buildkit | xargs -r docker rm -f || true

log "Pruning normal Docker objects, including named volumes"
docker system prune -af --volumes || true

log "Stopping Docker and containerd"
sudo systemctl stop docker docker.socket containerd || true

log "Ensuring mount source and target directories exist"
sudo mkdir -p "$VOLUME_SRC"
sudo mkdir -p "$VOLUME_DST"

log "Unmounting Docker volumes target if mounted"
if findmnt -rno TARGET "$VOLUME_DST" >/dev/null 2>&1; then
  sudo umount -l "$VOLUME_DST" || true
fi

log "Wiping app data safely"
rm -rf "$APP_DATA_DIR" || true
if [[ -d "$QDRANT_DIR" ]]; then
  sudo find "$QDRANT_DIR" -mindepth 1 -maxdepth 1 -exec rm -rf -- {} +
fi

log "Wiping Btrfs-backed Docker volume contents without deleting the mount source"
cleanup_dir_contents "$VOLUME_SRC"

log "Recreating mount target"
sudo mkdir -p "$VOLUME_DST"

log "Reloading systemd mount units"
sudo systemctl daemon-reload

log "Mounting Docker volumes target from fstab"
sudo mount "$VOLUME_DST"

log "Verifying mount"
findmnt "$VOLUME_DST"

log "Starting containerd and Docker"
sudo systemctl reset-failed docker || true
sudo systemctl start containerd
sudo systemctl start docker

log "Verifying Docker is up"
docker info >/dev/null

log "Recreating clean buildx builder"
if docker buildx inspect default >/dev/null 2>&1; then
  docker buildx use default >/dev/null 2>&1 || true
else
  docker buildx create --name default --use >/dev/null
fi
docker buildx inspect --bootstrap >/dev/null

log "Done"
