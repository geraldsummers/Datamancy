#!/usr/bin/env bash
# nuclear option:
set -u
sudo echo
# Stop containers
docker ps -q | xargs -r docker stop -t 0 || true
docker ps -aq | xargs -r docker rm -f || true

# Remove builders / cache
docker buildx prune -af || true
docker buildx rm --all-inactive || true

# Explicitly remove any buildx/buildkit containers that survived
docker ps -aq --filter name=buildx_buildkit | xargs -r docker rm -f || true
docker ps -aq --filter name=buildkit | xargs -r docker rm -f || true

# Remove normal Docker objects
docker system prune -af --volumes || true

# Stop daemons before deleting engine state
sudo systemctl stop docker containerd || true

# Remove engine state
sudo rm -rf /mnt/raid/docker

# Remove client-side state
rm -rf ~/.docker

# Optional: remove standalone containerd state too, only if you truly want a reset
sudo rm -rf /var/lib/containerd /run/containerd

# Your app data
sudo rm -rf ~/datamancy
sudo rm -rf /mnt/vectors/qdrant/*

sudo systemctl start containerd docker || true

# Recreate a clean buildx builder
docker buildx create --use >/dev/null
docker buildx inspect --bootstrap >/dev/null
